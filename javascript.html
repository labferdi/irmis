let time = 15;

if (time < 10) {
greeting = "Good morning";
} else if (time < 20) {
greeting = "Good day";
} else {
greeting = "Good evening";
}

console.log( greeting );
// Good day

let answer = 'a';

switch (answer) {
case 'a':
greeting = "Good morning";
break;
case 'b':
greeting = "Good day";
break;
default:
greeting = "Good evening";
}

console.log( greeting );
// Good morning
