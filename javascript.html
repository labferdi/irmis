Variable

Variable digunakan untuk menyimpan data

Numeric
// int

let a_int = 5;

// int (negative)

let b_int = -5;

// string int

let c_string = "5";

// float

let d_float = 3.14;

String
// string

let str_int = "ini string";

// string + int

let str_int = "ini string dan " + 5 + " digabung";

Array

let category = ['tagihan', 'pakaian', 'makanan'];

console.log( category.length );
// 3

console.log( category[0] );
// tagihan

console.log( category.indexOf('pakaian') );
// 1

console.log( category.indexOf('lorem') );
// -1

console.log( category.join(', ') );
// tagihan, pakaian, makanan

Object

let promo = {
name : 'promo oktober 2.22',
minimum_price : 100000,
discount : 0.5,
category : ['tagihan', 'pakaian', 'makanan']
}

console.log( promo.name );
// promo oktober 2.22

console.log( promo.category[0] );
// tagihan

console.log( promo.category.join(', ') );
// tagihan, pakaian, makanan

Comparison

if-else

let time = 15;
let greeting;

if(time < 10) {
greeting = "Good morning";
} else if(time < 20) {
greeting = "Good day";
} else {
greeting = "Good evening";
}

console.log( greeting );
// Good day

switch-case

let answer = "a";
let greeting;

switch(answer) {
case 'a':
greeting = "Good morning";
break;
case 'b':
greeting = "Good day";
break;
default:
greeting = "Good evening";
}

console.log( greeting );
// Good morning

Looping

for of

Looping untuk array variable

let category = ['tagihan', 'pakaian', 'makanan'];

for( let x of category ) {
console.log( x );
}

// tagihan
// pakaian
// makanan

for in

Looping untuk object variable

let promo = {
name : 'promo oktober 2.22',
minimum_price : 100000,
discount : 0.5,
}

for( let x in promo ) {
console.log( x, '=>', promo[x] );
}

// name => promo oktober 2.22
// minimum_price => 100000
// discount => 0.5
